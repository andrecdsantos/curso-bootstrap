                    let projectBoxes = document.querySelectorAll('.project-box');

                        filterBtn.addEventListener('click', function () {
                            // Remova a classe 'active' de todos os botões
                            filterBtns.forEach(function (btn) {
                                btn.classList.remove('active');
                            });

                            // Adicione a classe 'active' apenas ao botão clicado
                            filterBtn.classList.add('active');

                            // Obtém a categoria do botão clicado
                            let category = filterBtn.id.replace('-btn', '');

                            // Oculta os projetos

                            projectBoxes.forEach(function (projectBox) {

                                    // Mostra os projetos
                                    projectBox.style.display = 'block';
