        document.addEventListener('DOMContentLoaded', function () {
          let navLinks = document.querySelectorAll('.nav-link');
          let principal = document.querySelector('.principal');
          let produtos = document.querySelector('.produtos');
          let contato = document.querySelector('.contato');
          let sobre = document.querySelector('.sobre');

          navLinks.forEach(function (navLink) {
              navLink.addEventListener('click', function(event) {
                  event.preventDefault();  // Impede o comportamento padrão do link
                  if (navLink.id === 'homeMenu') {
                      scrollToSection(principal);
                  } else if (navLink.id === 'produtosMenu') {
                      scrollToSection(produtos);
                  } else if (navLink.id === 'contatoMenu') {
                      scrollToSection(contato);
                  } else if (navLink.id === 'sobreMenu') {
                      scrollToSection(sobre);
                  }
              });
          });

          function scrollToSection(section) {
              // Calcula a posição da parte superior da seção
              let sectionTop = section.offsetTop;

              // Rola suavemente para a posição da seção
              window.scrollTo({
                  top: sectionTop,
                  behavior: 'smooth'
              });
          }
      });
